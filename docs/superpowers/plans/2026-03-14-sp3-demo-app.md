# examples/refund_agent/agent.py
"""AI Refund Agent — does what Pine AI does, in 50 lines.

Usage:
    # CLI mode
    python -m examples.refund_agent.agent "+18001234567" \
        --task "Cancel my Comcast internet subscription" \
        --name "Alice Chen" --account "98765"

    # Or via call-use CLI
    call-use dial "+18001234567" \
        -i "Cancel my internet subscription. Be firm but polite. Get confirmation number." \
        -u '{"name": "Alice Chen", "account_number": "98765"}'
"""
import asyncio
import json
import sys

from call_use import CallAgent, CallEvent


REFUND_INSTRUCTIONS = """
{task}

Strategy:
1. Navigate the phone menu to reach a human agent (billing/cancellation dept)
2. Clearly state the request: cancellation or refund
3. If offered a discount or retention deal, decline politely and insist on cancellation
4. Get a confirmation number or reference ID before ending the call
5. If asked why, say "I've found a better option that fits my needs"
6. Be firm but polite throughout

Important:
- Do NOT accept retention offers unless explicitly told to
- Do NOT agree to any charges or fees without approval
- Get the agent's name and confirmation number before hanging up
""".strip()


def _on_event(event: CallEvent):
    """Print call events to stderr."""
    t = event.type.value
    if t == "transcript":
        speaker = event.data.get("speaker", "?")
        text = event.data.get("text", "")
        icon = "🤖" if speaker == "agent" else "👤"
        print(f"  {icon} [{speaker}] {text}", file=sys.stderr)
    elif t == "state_change":
        print(f"  📞 {event.data.get('from', '?')} → {event.data.get('to', '?')}", file=sys.stderr)


def _on_approval(details: dict) -> str:
    """Ask human for approval on sensitive actions."""
    print(f"\n  ⚠️  APPROVAL NEEDED: {details.get('details', '')}", file=sys.stderr)
    response = input("  Approve? (y/n): ").strip().lower()
    return "approved" if response == "y" else "rejected"


async def run_refund_agent(
    phone: str,
    task: str,
    name: str | None = None,
    account: str | None = None,
    approval: bool = True,
) -> dict:
    """Run the refund agent and return structured outcome."""
    user_info = {}
    if name:
        user_info["name"] = name
    if account:
        user_info["account_number"] = account

    agent = CallAgent(
        phone=phone,
        instructions=REFUND_INSTRUCTIONS.format(task=task),
        user_info=user_info,
        approval_required=approval,
        on_event=_on_event,
        on_approval=_on_approval if approval else None,
    )

    print(f"\n📞 Calling {phone}...", file=sys.stderr)
    print(f"📋 Task: {task}", file=sys.stderr)
    print("─" * 50, file=sys.stderr)

    outcome = await agent.call()

    print("─" * 50, file=sys.stderr)
    print(f"✅ Result: {outcome.disposition.value}", file=sys.stderr)
    print(f"⏱  Duration: {outcome.duration_seconds:.0f}s", file=sys.stderr)
    print(f"💬 Transcript: {len(outcome.transcript)} turns", file=sys.stderr)

    return outcome.model_dump(mode="json")


def main():
    import argparse
    parser = argparse.ArgumentParser(description="AI Refund Agent — Pine AI in 50 lines")
    parser.add_argument("phone", help="Phone number to call (E.164)")
    parser.add_argument("--task", "-t", required=True, help="What to do")
    parser.add_argument("--name", "-n", help="Your name for verification")
    parser.add_argument("--account", "-a", help="Account number")
    parser.add_argument("--no-approval", action="store_true", help="Skip approval prompts")
    args = parser.parse_args()

    result = asyncio.run(run_refund_agent(
        phone=args.phone,
        task=args.task,
        name=args.name,
        account=args.account,
        approval=not args.no_approval,
    ))
    print(json.dumps(result, indent=2))


if __name__ == "__main__":
    main()
```

# examples/refund_agent/web.py
"""Web UI for AI Refund Agent — live transcript streaming via SSE.

Usage:
    pip install call-use fastapi uvicorn
    python -m examples.refund_agent.web
    # Open http://localhost:8000
"""
import asyncio
import json
import os
from pathlib import Path

from fastapi import FastAPI
from fastapi.responses import HTMLResponse, StreamingResponse
from fastapi.staticfiles import StaticFiles
from pydantic import BaseModel

from call_use import CallAgent, CallEvent

app = FastAPI(title="AI Refund Agent Demo")

STATIC_DIR = Path(__file__).parent / "static"


class RefundRequest(BaseModel):
    phone: str
    task: str
    name: str = ""
    account: str = ""


@app.get("/", response_class=HTMLResponse)
async def index():
    return (STATIC_DIR / "index.html").read_text()


@app.post("/api/call")
async def start_call(req: RefundRequest):
    """Start a call and stream events via SSE."""
    event_queue: asyncio.Queue = asyncio.Queue()

    def on_event(event: CallEvent):
        asyncio.get_event_loop().call_soon_threadsafe(
            event_queue.put_nowait,
            event.model_dump(mode="json"),
        )

    user_info = {}
    if req.name:
        user_info["name"] = req.name
    if req.account:
        user_info["account_number"] = req.account

    instructions = f"""
{req.task}

Strategy:
1. Navigate phone menu to reach billing/cancellation
2. State the request clearly
3. Decline retention offers, insist on cancellation
4. Get confirmation number before ending
5. Be firm but polite
""".strip()

    agent = CallAgent(
        phone=req.phone,
        instructions=instructions,
        user_info=user_info,
        approval_required=False,
        on_event=on_event,
    )

    async def event_stream():
        call_task = asyncio.create_task(agent.call())

        while not call_task.done():
            try:
                event = await asyncio.wait_for(event_queue.get(), timeout=1.0)
                yield f"data: {json.dumps(event)}\n\n"
            except asyncio.TimeoutError:
                yield f"data: {json.dumps({'type': 'heartbeat'})}\n\n"

        # Drain remaining events
        while not event_queue.empty():
            event = event_queue.get_nowait()
            yield f"data: {json.dumps(event)}\n\n"

        outcome = call_task.result()
        yield f"data: {json.dumps({'type': 'final_outcome', 'data': outcome.model_dump(mode='json')})}\n\n"

    return StreamingResponse(event_stream(), media_type="text/event-stream")


def main():
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)


if __name__ == "__main__":
    main()
```
