# SP1: CLI + MCP Server Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Any AI agent (Claude Code, Codex, LangChain, CrewAI) can make phone calls via CLI or MCP tool with zero Python code.

**Architecture:** CLI wraps `CallAgent` SDK synchronously. MCP server dispatches calls asynchronously via LiveKit API (non-blocking `dial` returns `task_id`, with `status`/`cancel`/`result` tools for lifecycle management). Both produce structured JSON output. CLI streams events to stderr for human observability.

**Tech Stack:** Click (CLI framework), mcp[server] (MCP SDK), existing call_use SDK

```python
# call_use/cli.py
"""call-use CLI — agent-native interface for making outbound calls."""

import asyncio
import json
import sys

import click

from call_use.models import CallEvent, DispositionEnum


def _event_printer(event: CallEvent):
    """Print events to stderr for real-time observability."""
    if event.type.value == "transcript":
        speaker = event.data.get("speaker", "?")
        text = event.data.get("text", "")
        click.echo(f"  [{speaker}] {text}", err=True)
    elif event.type.value == "state_change":
        new_state = event.data.get("to", "?")
        click.echo(f"  state: {new_state}", err=True)
    elif event.type.value == "approval_request":
        details = event.data.get("details", "")
        click.echo(f"  APPROVAL NEEDED: {details}", err=True)


def _stdin_approval_handler(details: str) -> bool:
    """Interactive approval handler — prompts user on stdin."""
    click.echo(f"  APPROVAL NEEDED: {details}", err=True)
    response = click.prompt("  Approve? [y/n]", type=click.Choice(["y", "n"]), err=True)
    return response == "y"


def _run_call(
    phone: str,
    instructions: str,
    user_info: dict | None = None,
    caller_id: str | None = None,
    voice_id: str | None = None,
    timeout: int = 600,
    approval_required: bool = False,
) -> dict:
    """Run a call synchronously via CallAgent. Returns outcome dict."""
    from call_use.sdk import CallAgent

    kwargs = dict(
        phone=phone,
        instructions=instructions,
        user_info=user_info,
        caller_id=caller_id,
        voice_id=voice_id,
        approval_required=approval_required,
        timeout_seconds=timeout,
        on_event=_event_printer,
    )
    # SDK raises ValueError if approval_required=True without on_approval callback
    if approval_required:
        kwargs["on_approval"] = _stdin_approval_handler

    agent = CallAgent(**kwargs)
    outcome = asyncio.run(agent.call())
    return outcome.model_dump(mode="json")


@click.group()
@click.version_option(package_name="call-use")
def main():
    """call-use: give your AI agent the ability to make phone calls."""
    pass


@main.command()
@click.argument("phone")
@click.option("--instructions", "-i", required=True, help="What the agent should do on the call.")
@click.option("--user-info", "-u", default=None, help="JSON dict of context for the agent.")
@click.option("--caller-id", default=None, help="Outbound caller ID (E.164).")
@click.option("--voice-id", default=None, help="TTS voice (alloy, echo, fable, onyx, nova, shimmer).")
@click.option("--timeout", default=600, type=int, help="Max call duration in seconds.")
@click.option("--approval-required", is_flag=True, default=False, help="Require approval for sensitive actions.")
def dial(phone, instructions, user_info, caller_id, voice_id, timeout, approval_required):
    """Make an outbound phone call.

    PHONE: Target phone number in E.164 format (e.g., +18001234567)

    Events stream to stderr in real-time. Structured JSON result goes to stdout.

    \b
    Examples:
        call-use dial "+18001234567" -i "Ask about store hours"
        call-use dial "+18005551234" -i "Cancel subscription" -u '{"account": "12345"}'
    """
    parsed_user_info = None
    if user_info:
        try:
            parsed_user_info = json.loads(user_info)
        except json.JSONDecodeError:
            click.echo("Error: --user-info must be valid JSON", err=True)
            sys.exit(2)

    click.echo(f"Calling {phone}...", err=True)
    try:
        result = _run_call(
            phone=phone,
            instructions=instructions,
            user_info=parsed_user_info,
            caller_id=caller_id,
            voice_id=voice_id,
            timeout=timeout,
            approval_required=approval_required,
        )
    except Exception as e:
        click.echo(f"Error: {e}", err=True)
        sys.exit(2)

    # Structured JSON to stdout (agent-parseable)
    click.echo(json.dumps(result, indent=2))

    # Exit codes:
    #   0 = expected outcome (completed, voicemail, no_answer, busy)
    #   1 = actual failure (failed, timeout, cancelled)
    #   2 = input error (handled above)
    disposition = result.get("disposition")
    expected_outcomes = {"completed", "voicemail", "no_answer", "busy"}
    if disposition not in expected_outcomes:
        sys.exit(1)
```

- [ ] **Step 3: Implement MCP server**

**IMPORTANT design note:** MCP servers over stdio process one tool call at a time.
If `dial` blocks for the full call duration (1-10 minutes), then `status` and `cancel`
tools become unusable since the server is frozen. Therefore, `dial` MUST be non-blocking:
it dispatches the call via LiveKit API and returns a `task_id` immediately. Use `status`
to poll call state, `cancel` to abort, and `result` to retrieve the final outcome.

```python
# call_use/mcp_server.py
"""call-use MCP server — expose phone calling as tools for AI agents.

Architecture: Non-blocking async design.
- dial: dispatches call via LiveKit API, returns task_id immediately
- status: polls room metadata for call state
- cancel: sends cancel command via data channel
- result: retrieves final CallOutcome from room metadata after completion

This avoids the stdio MCP single-request-at-a-time bottleneck.
"""

import json
import logging
import os
import uuid

from livekit.api import LiveKitAPI
from mcp.server import FastMCP

logger = logging.getLogger(__name__)

mcp = FastMCP(
    "call-use",
    description="Give your AI agent the ability to make phone calls. The 'browser-use' for phones.",
)


def _get_livekit_api() -> LiveKitAPI:
    """Create LiveKitAPI client from environment."""
    return LiveKitAPI(
        url=os.environ["LIVEKIT_URL"],
        api_key=os.environ["LIVEKIT_API_KEY"],
        api_secret=os.environ["LIVEKIT_API_SECRET"],
    )


async def _do_dial(
    phone: str,
    instructions: str,
    user_info: dict | None = None,
    caller_id: str | None = None,
    voice_id: str | None = None,
    timeout: int = 600,
) -> dict:
    """Internal: dispatch a call via LiveKit SIP and return immediately with task_id.

    Creates a LiveKit room, stores call config as room metadata, and dispatches
    a SIP participant into the room. The call-use agent worker picks up the room
    and runs the call autonomously. Does NOT block for call completion.
    """
    api = _get_livekit_api()
    task_id = f"call-{uuid.uuid4().hex[:12]}"

    metadata = json.dumps({
        "phone": phone,
        "instructions": instructions,
        "user_info": user_info,
        "caller_id": caller_id,
        "voice_id": voice_id,
        "timeout": timeout,
        "state": "dispatching",
    })

    # Create room for this call
    await api.room.create_room(
        name=task_id,
        metadata=metadata,
        empty_timeout=timeout + 60,
    )

    # Dispatch SIP participant into the room
    await api.sip.create_sip_participant(
        room_name=task_id,
        sip_trunk_id=os.environ.get("SIP_TRUNK_ID", ""),
        sip_call_to=phone,
        participant_identity=task_id,
    )

    return {"task_id": task_id, "status": "dispatched"}


async def _do_status(task_id: str) -> dict:
    """Internal: poll room metadata for call state."""
    api = _get_livekit_api()
    rooms = await api.room.list_rooms(names=[task_id])
    if not rooms.rooms:
        return {"task_id": task_id, "error": "call not found"}

    room = rooms.rooms[0]
    metadata = json.loads(room.metadata) if room.metadata else {}
    return {
        "task_id": task_id,
        "state": metadata.get("state", "unknown"),
        "duration_seconds": metadata.get("duration_seconds", 0.0),
    }


async def _do_result(task_id: str) -> dict:
    """Internal: retrieve final CallOutcome from room metadata."""
    api = _get_livekit_api()
    rooms = await api.room.list_rooms(names=[task_id])
    if not rooms.rooms:
        return {"task_id": task_id, "error": "call not found"}

    room = rooms.rooms[0]
    metadata = json.loads(room.metadata) if room.metadata else {}

    if metadata.get("state") not in ("completed", "failed", "voicemail", "no_answer", "busy", "timeout"):
        return {"task_id": task_id, "error": "call still in progress", "state": metadata.get("state")}

    return metadata.get("outcome", {"task_id": task_id, "error": "no outcome available"})


@mcp.tool()
async def dial(
    phone: str,
    instructions: str,
    user_info: str | None = None,
    caller_id: str | None = None,
    voice_id: str | None = None,
    timeout: int = 600,
) -> str:
    """Dispatch an outbound phone call via AI agent. Returns immediately with a task_id.

    The call runs asynchronously. Use 'status' to poll progress, 'cancel' to abort,
    and 'result' to retrieve the final outcome once the call completes.

    Args:
        phone: Target phone number in E.164 format (e.g., +18001234567). US/Canada only.
        instructions: What the agent should accomplish on the call (e.g., "Cancel my subscription. Account #12345.").
        user_info: Optional JSON string with context for the agent (e.g., '{"name": "Alice", "account": "12345"}').
        caller_id: Optional outbound caller ID in E.164 format.
        voice_id: TTS voice: alloy, echo, fable, onyx, nova, or shimmer. Default: alloy.
        timeout: Max call duration in seconds. Default: 600.

    Returns:
        JSON with: task_id and status ("dispatched"). Use 'status' tool to poll progress.
    """
    parsed_info = None
    if user_info:
        try:
            parsed_info = json.loads(user_info)
        except json.JSONDecodeError:
            return json.dumps({"error": "user_info must be valid JSON"})

    try:
        result = await _do_dial(
            phone=phone,
            instructions=instructions,
            user_info=parsed_info,
            caller_id=caller_id,
            voice_id=voice_id,
            timeout=timeout,
        )
        return json.dumps(result, indent=2)
    except Exception as e:
        return json.dumps({"error": str(e)})


@mcp.tool()
async def status(task_id: str) -> str:
    """Check the current state of a phone call.

    Args:
        task_id: The task_id returned by the 'dial' tool.

    Returns:
        JSON with: task_id, state (dispatching/in_progress/completed/failed/...),
        and duration_seconds.
    """
    try:
        result = await _do_status(task_id)
        return json.dumps(result, indent=2)
    except Exception as e:
        return json.dumps({"error": str(e), "task_id": task_id})


@mcp.tool()
async def cancel(task_id: str) -> str:
    """Cancel an active phone call.

    Args:
        task_id: The task_id of the call to cancel.

    Returns:
        JSON with cancellation status.
    """
    try:
        api = _get_livekit_api()
        # Send cancel command via data channel to the agent in the room
        await api.room.send_data(
            room=task_id,
            data=json.dumps({"command": "cancel"}).encode(),
            topic="call-control",
        )
        return json.dumps({"task_id": task_id, "status": "cancel_requested"})
    except Exception as e:
        return json.dumps({"error": str(e), "task_id": task_id})


@mcp.tool()
async def result(task_id: str) -> str:
    """Retrieve the final outcome of a completed phone call.

    Call this after 'status' shows the call has finished. Returns the full
    transcript, disposition, and events.

    Args:
        task_id: The task_id returned by the 'dial' tool.

    Returns:
        JSON with: task_id, disposition (completed/failed/no_answer/busy/voicemail/timeout),
        duration_seconds, transcript (list of {speaker, text}), and events.
        Returns error if call is still in progress.
    """
    try:
        outcome = await _do_result(task_id)
        return json.dumps(outcome, indent=2)
    except Exception as e:
        return json.dumps({"error": str(e), "task_id": task_id})


def main():
    """CLI entrypoint for call-use MCP server (stdio transport)."""
    mcp.run(transport="stdio")


if __name__ == "__main__":
    main()
```

```python
# examples/langchain_tool.py
"""Use call-use as a LangChain tool.

pip install call-use langchain-core
"""
import json
import subprocess

from langchain_core.tools import tool


@tool
def phone_call(phone: str, instructions: str, user_info: str = "{}") -> str:
    """Make a phone call via AI agent. Returns JSON with transcript and outcome.

    Args:
        phone: Target phone number in E.164 format (e.g., +18001234567)
        instructions: What to accomplish on the call
        user_info: JSON string with context (e.g., '{"name": "Alice"}')
    """
    result = subprocess.run(
        ["call-use", "dial", phone, "-i", instructions, "-u", user_info],
        capture_output=True, text=True, timeout=660,
    )
    if result.returncode == 2:
        return json.dumps({"error": f"Input error: {result.stderr.strip()}"})
    if result.returncode != 0 and not result.stdout.strip():
        return json.dumps({"error": f"Call failed (exit {result.returncode}): {result.stderr.strip()}"})
    return result.stdout
```
