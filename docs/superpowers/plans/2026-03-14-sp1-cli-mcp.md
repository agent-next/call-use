# SP1: CLI + MCP Server Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Any AI agent (Claude Code, Codex, LangChain, CrewAI) can make phone calls via CLI or MCP tool with zero Python code.

**Architecture:** CLI wraps `CallAgent` SDK. MCP server exposes tools over stdio transport. Both produce structured JSON output. CLI streams events to stderr for human observability.

**Tech Stack:** Click (CLI framework), mcp[server] (MCP SDK), existing call_use SDK

# call_use/cli.py
"""call-use CLI — agent-native interface for making outbound calls."""

import asyncio
import json
import sys

import click

from call_use.models import CallEvent, DispositionEnum


def _event_printer(event: CallEvent):
    """Print events to stderr for real-time observability."""
    if event.type.value == "transcript":
        speaker = event.data.get("speaker", "?")
        text = event.data.get("text", "")
        click.echo(f"  [{speaker}] {text}", err=True)
    elif event.type.value == "state_change":
        new_state = event.data.get("to", "?")
        click.echo(f"  state: {new_state}", err=True)
    elif event.type.value == "approval_request":
        details = event.data.get("details", "")
        click.echo(f"  APPROVAL NEEDED: {details}", err=True)


def _run_call(
    phone: str,
    instructions: str,
    user_info: dict | None = None,
    caller_id: str | None = None,
    voice_id: str | None = None,
    timeout: int = 600,
    approval_required: bool = False,
) -> dict:
    """Run a call synchronously via CallAgent. Returns outcome dict."""
    from call_use.sdk import CallAgent

    agent = CallAgent(
        phone=phone,
        instructions=instructions,
        user_info=user_info,
        caller_id=caller_id,
        voice_id=voice_id,
        approval_required=approval_required,
        timeout_seconds=timeout,
        on_event=_event_printer,
    )
    outcome = asyncio.run(agent.call())
    return outcome.model_dump(mode="json")


@click.group()
@click.version_option(package_name="call-use")
def main():
    """call-use: give your AI agent the ability to make phone calls."""
    pass


@main.command()
@click.argument("phone")
@click.option("--instructions", "-i", required=True, help="What the agent should do on the call.")
@click.option("--user-info", "-u", default=None, help="JSON dict of context for the agent.")
@click.option("--caller-id", default=None, help="Outbound caller ID (E.164).")
@click.option("--voice-id", default=None, help="TTS voice (alloy, echo, fable, onyx, nova, shimmer).")
@click.option("--timeout", default=600, type=int, help="Max call duration in seconds.")
@click.option("--approval-required", is_flag=True, default=False, help="Require approval for sensitive actions.")
def dial(phone, instructions, user_info, caller_id, voice_id, timeout, approval_required):
    """Make an outbound phone call.

    PHONE: Target phone number in E.164 format (e.g., +18001234567)

    Events stream to stderr in real-time. Structured JSON result goes to stdout.

    \b
    Examples:
        call-use dial "+18001234567" -i "Ask about store hours"
        call-use dial "+18005551234" -i "Cancel subscription" -u '{"account": "12345"}'
    """
    parsed_user_info = None
    if user_info:
        try:
            parsed_user_info = json.loads(user_info)
        except json.JSONDecodeError:
            click.echo("Error: --user-info must be valid JSON", err=True)
            sys.exit(2)

    click.echo(f"Calling {phone}...", err=True)
    try:
        result = _run_call(
            phone=phone,
            instructions=instructions,
            user_info=parsed_user_info,
            caller_id=caller_id,
            voice_id=voice_id,
            timeout=timeout,
            approval_required=approval_required,
        )
    except Exception as e:
        click.echo(f"Error: {e}", err=True)
        sys.exit(2)

    # Structured JSON to stdout (agent-parseable)
    click.echo(json.dumps(result, indent=2))

    # Exit code: 0 = completed, 1 = any other disposition
    if result.get("disposition") != "completed":
        sys.exit(1)
```

# call_use/mcp_server.py
"""call-use MCP server — expose phone calling as tools for AI agents."""

import asyncio
import json
import logging

from mcp.server import FastMCP

from call_use.models import CallEvent

logger = logging.getLogger(__name__)

mcp = FastMCP(
    "call-use",
    description="Give your AI agent the ability to make phone calls. The 'browser-use' for phones.",
)


async def _do_dial(
    phone: str,
    instructions: str,
    user_info: dict | None = None,
    caller_id: str | None = None,
    voice_id: str | None = None,
    timeout: int = 600,
) -> dict:
    """Internal: run a call and return outcome dict."""
    from call_use.sdk import CallAgent

    agent = CallAgent(
        phone=phone,
        instructions=instructions,
        user_info=user_info,
        caller_id=caller_id,
        voice_id=voice_id,
        approval_required=False,
        timeout_seconds=timeout,
    )
    outcome = await agent.call()
    return outcome.model_dump(mode="json")


@mcp.tool()
async def dial(
    phone: str,
    instructions: str,
    user_info: str | None = None,
    caller_id: str | None = None,
    voice_id: str | None = None,
    timeout: int = 600,
) -> str:
    """Make an outbound phone call via AI agent.

    The agent dials the phone number, follows your instructions, navigates IVR menus,
    talks to humans, and returns a structured result with transcript and disposition.

    Args:
        phone: Target phone number in E.164 format (e.g., +18001234567). US/Canada only.
        instructions: What the agent should accomplish on the call (e.g., "Cancel my subscription. Account #12345.").
        user_info: Optional JSON string with context for the agent (e.g., '{"name": "Alice", "account": "12345"}').
        caller_id: Optional outbound caller ID in E.164 format.
        voice_id: TTS voice: alloy, echo, fable, onyx, nova, or shimmer. Default: alloy.
        timeout: Max call duration in seconds. Default: 600.

    Returns:
        JSON with: task_id, disposition (completed/failed/no_answer/busy/voicemail/timeout),
        duration_seconds, transcript (list of {speaker, text}), and events.
    """
    parsed_info = None
    if user_info:
        try:
            parsed_info = json.loads(user_info)
        except json.JSONDecodeError:
            return json.dumps({"error": "user_info must be valid JSON"})

    try:
        result = await _do_dial(
            phone=phone,
            instructions=instructions,
            user_info=parsed_info,
            caller_id=caller_id,
            voice_id=voice_id,
            timeout=timeout,
        )
        return json.dumps(result, indent=2)
    except Exception as e:
        return json.dumps({"error": str(e)})


@mcp.tool()
async def cancel(call_id: str) -> str:
    """Cancel an active phone call.

    Args:
        call_id: The task_id of the call to cancel.

    Returns:
        JSON with cancellation status.
    """
    # TODO: implement via LiveKit API room lookup + cancel command
    return json.dumps({"status": "cancel not yet implemented", "call_id": call_id})


def main():
    """CLI entrypoint for call-use MCP server (stdio transport)."""
    mcp.run(transport="stdio")


if __name__ == "__main__":
    main()
```

"""Use call-use as a LangChain tool.

pip install call-use langchain-core
"""
import asyncio
import json
import subprocess

from langchain_core.tools import tool


@tool
def phone_call(phone: str, instructions: str, user_info: str = "{}") -> str:
    """Make a phone call via AI agent. Returns JSON with transcript and outcome.

    Args:
        phone: Target phone number in E.164 format (e.g., +18001234567)
        instructions: What to accomplish on the call
        user_info: JSON string with context (e.g., '{"name": "Alice"}')
    """
    result = subprocess.run(
        ["call-use", "dial", phone, "-i", instructions, "-u", user_info],
        capture_output=True, text=True, timeout=660,
    )
    return result.stdout


# Usage with any LangChain agent:
